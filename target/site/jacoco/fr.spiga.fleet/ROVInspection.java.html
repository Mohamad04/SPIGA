package fr.spiga.fleet;

import fr.spiga.core.Position3D;

/**
 * ROV d'inspection téléopéré spécialisé en inspection sous-marine.
 * Extension du véhicule sous-marin avec caméras et bras manipulateur.
 * 
 * @author SPIGA Team
 * @version 1.0
 */
public class ROVInspection extends VehiculeSousMarin {

    private static final double VITESSE_MAX_DEFAUT = 5.0; // m/s (lent, précis)
    private static final double AUTONOMIE_MAX_DEFAUT = 8.0; // heures

    /** Nombre de caméras embarquées */
    private final int nombreCameras;

    /** Indique si le bras manipulateur est actif */
    private boolean brasActif;

    /**
     * Constructeur avec paramètres par défaut.
     * 
     * @param position la position initiale
     */
    public ROVInspection(Position3D position) {
        super(position, VITESSE_MAX_DEFAUT, AUTONOMIE_MAX_DEFAUT);
        this.nombreCameras = 4;
        this.brasActif = false;
    }

    public int getNombreCameras() {
        return nombreCameras;
    }

    public boolean isBrasActif() {
        return brasActif;
    }

    /**
     * Active le bras manipulateur.
     */
    public void activerBras() {
        brasActif = true;
    }

    /**
     * Désactive le bras manipulateur.
     */
    public void desactiverBras() {
        brasActif = false;
    }

    /**
     * Effectue une inspection visuelle.
     * 
     * @return le résultat de l'inspection
     */
    public String effectuerInspection() {
        return String.format("Inspection à %.1fm avec %d caméras - Bras: %s",
                getProfondeur(), nombreCameras, brasActif ? "actif" : "inactif");
    }

    @Override
    public String getType() {
        return "ROVInspection";
    }
}
