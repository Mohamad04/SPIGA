package fr.spiga.fleet;

import fr.spiga.core.Position3D;

/**
 * AUV de reconnaissance spécialisé en exploration sous-marine.
 * Extension du véhicule sous-marin avec capacités de cartographie.
 * 
 * @author SPIGA Team
 * @version 1.0
 */
public class AUVReconnaissance extends VehiculeSousMarin {

    private static final double VITESSE_MAX_DEFAUT = 12.0; // m/s (plus rapide)
    private static final double AUTONOMIE_MAX_DEFAUT = 15.0; // heures (longue durée)

    /** Portée du sonar en mètres */
    private final double porteeSonar;

    /**
     * Constructeur avec paramètres par défaut.
     * 
     * @param position la position initiale
     */
    public AUVReconnaissance(Position3D position) {
        super(position, VITESSE_MAX_DEFAUT, AUTONOMIE_MAX_DEFAUT);
        this.porteeSonar = 5000.0; // 5 km
    }

    public double getPorteeSonar() {
        return porteeSonar;
    }

    /**
     * Effectue un scan sonar de la zone.
     * 
     * @return le résultat du scan
     */
    public String effectuerScanSonar() {
        return String.format("Scan sonar à profondeur %.1fm - Portée: %.0fm",
                getProfondeur(), porteeSonar);
    }

    @Override
    public String getType() {
        return "AUVReconnaissance";
    }
}
