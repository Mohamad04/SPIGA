package fr.spiga.fleet;

import fr.spiga.core.Position3D;

/**
 * ASV de patrouille spécialisé en surveillance maritime de surface.
 * Extension du véhicule de surface avec capacités de détection.
 * 
 * @author SPIGA Team
 * @version 1.0
 */
public class ASVPatrouille extends VehiculeSurface {

    private static final double VITESSE_MAX_DEFAUT = 15.0; // m/s (plus rapide)
    private static final double AUTONOMIE_MAX_DEFAUT = 20.0; // heures (très longue durée)

    /** Portée du radar en mètres */
    private final double porteeRadar;

    /**
     * Constructeur avec paramètres par défaut.
     * 
     * @param position la position initiale
     */
    public ASVPatrouille(Position3D position) {
        super(position, VITESSE_MAX_DEFAUT, AUTONOMIE_MAX_DEFAUT);
        this.porteeRadar = 10000.0; // 10 km
    }

    public double getPorteeRadar() {
        return porteeRadar;
    }

    /**
     * Effectue un scan radar de la zone.
     * 
     * @return le résultat du scan
     */
    public String effectuerScanRadar() {
        return String.format("Scan radar - Portée: %.0fm", porteeRadar);
    }

    @Override
    public String getType() {
        return "ASVPatrouille";
    }
}
